# 풀이
Mo's Algorithm을 이용해 쿼리를 정렬한 뒤, 쿼리를 처리해나가면서 색깔별 모자 개수를 관리한다. 색깔의 개수가 적어서 매 쿼리마다 과반수의 모자가 있는지 모두 돌면서 확인해도 충분하다.

# 코드
```c++
#include <bits/stdc++.h>
#define fio cin.tie(0), ios_base::sync_with_stdio(0)

using namespace std;

int n, sn, c, h[300007], m, lp=1, rp, ans[10007], cnt[10007], mi;
struct query{
    int l, r, ix;
    bool operator <(const query &q) const{
        return r/sn == q.r/sn ? l < q.l : r/sn < q.r/sn;
    }
} q[10007];

void f(int idx, int add){
    cnt[h[idx]] += add;
}

int main(){
    fio;
    cin >> n >> c;
    sn = (int)sqrt(n);
    for(int i=0; i<n; i++) cin >> h[i];
    cin >> m;
    for(int i=0, a, b; i<m; i++){
        cin >> a >> b;
        q[i].l = --a;
        q[i].r = --b;
        q[i].ix = i;
    }
    sort(q, q+m);
    for(int i=0; i<m; i++){
        auto [l, r, ix] = q[i];
        while(lp < l) f(lp++, -1);
        while(lp > l) f(--lp, 1);
        while(rp < r) f(++rp, 1);
        while(rp > r) f(rp--, -1);
        mi = -1;
        for(int j=1; j<=c; j++){
            if(cnt[j] > (r-l+1)/2) mi = j;
        }
        ans[ix] = mi;
    }
    for(int i=0; i<m; i++){
        if(ans[i] == -1){
            cout << "no\n";
            continue;
        }
        cout << "yes " << ans[i] << "\n";
    }
}
```
